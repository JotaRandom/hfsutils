\subsection{Commands}
\begin{itemize}
    \item hformat - Format HFS volumes
    \item hmount - Mount HFS volumes
    \item humount - Unmount HFS volumes
    \item hls - List directory contents
    \item hcopy - Copy files
    \item hmkdir - Create directories
    \item And more...
\end{itemize}
